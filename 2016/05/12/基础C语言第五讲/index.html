    练习1.1

    int a[2][3] = {{1,2,3},{4,5,6}};
int b[3][2] = {0};
for (int i = 0; i < 2; i ++) {
    for (int k = 0; k < 3; k ++) {
        b[k][i] = a[i][k];
    }
}
for (int i = 0; i < 3; i++) {
    for (int k = 0; k < 2; k ++) {
        printf("%d",b[i][k]);
    }
    printf("\n");
}

    练习1.2

     int a[3][4] = {{5,6,7,11}
                  ,{9,0,1,12}
                  ,{1,2,3,4}};
int max = 0;
int x = 0;
int y = 0;
for (int i = 0; i < 3; i ++) {
    for (int k = 0; k < 4; k ++) {
        if (a[i][k] > max) {
            max = a[i][k];
            x = i;
            y = k;
        }
    }
}
printf("max = %d",max);
printf("在第%d行，第%d列",x,y);
