    递归 练习1.4 打断点演示一下自己就好了


    // 5! = 5 4!

    // 4! = 4     3! …

    int func(int n)
{
    //    printf("正在计算%d的阶乘\n", n);

    // 0和1的阶乘都是1
    if (n == 0 || n == 1) {//（递归一定要有出口）

        return 1;
    }
    // 5! = 5 * 4!
    // 4! = 4 * 3! ...
    int result = n * func(n - 1);//(递归就是函数里面又调用了函数)

    return result;
}
