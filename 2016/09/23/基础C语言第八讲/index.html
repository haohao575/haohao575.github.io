一·数据的存储

字节 是最小的存储单位 一个字节 未一个存储单位（内存单元） 不同数据类型 存储的单位不同
内存是由多个连续的内存单元组成.
内存单元用于存储数据，每个内存单元占1个字节，1个字节 = 8个二进制位
内存单元地址：内存单元的编号，连续的。便于访问内存单元，就像门牌号一样。

二·内存的访问方式

直接访问 与 间接访问
直接访问 是访问内存空间的内容 如 a = 20,b = 2 * a
间接访问 是通过内存单元编号 以及 数据所占 字节数 访问数据的数据
指针就是间接方式访问的一种

三·指针变量

int * p = NULL; //初始值NULL恒等于 0
//int * 为整数指针 注意指针变量所占字节数 只与操作系统有关
//打印指针变量的值
printf("p = %p \n",p);

四·取址运算符 &


int a = 10;
int b = 20;
int pa = &a;//取出整形数a的地址
int pb = &b;//取出整形数b的地址
printf(“pa = %p,a = %d &a = %p\n”,pa,a,&a);
printf(“pb = %p,b = %d &b = %p\n”,pb,b,&b);

char p = str;

for (int i = 0; i < 6; i++) {

// “”常量字符串，存储在常量区，只能访问不能修改
// 定义指针指向常量区的字符串常量的首地址，指针p中只是存储的地址
char p = “hello”;
p = ‘a’;
// 可以访问字符和字符串
printf(“%s\n”, p);
printf(“%c\n”, p);
// 总结：指向数组的指针可以访问和修改数组元素；指向常量字符串的指针只能访问数组元素

指针与字符串数组


//指针数组 数组中的元素都是指针
char strs[3] = {“iOS”, “iPad”, “iMac”};
// strs本质是一维数组
// strs存储的数组元素是字符指针(字符指针指向常量字符串)
// strs[0] 第一个字符指针，指向常量区的”iOS”
// strs[1] 第二个字符指针，指向常量区的”iPad”
// strs[2] 第三个字符指针，指向常量区的”iMac”
// 访问
printf(“%s\n”, strs[0]);
printf(“%s\n”, strs[1]);
printf(“%s\n”, strs);
printf(“%s\n”, (strs + 1));
// 修改
strs[0] = “Symbian”; // strs[0]访问的是字符指针，指针的重新指向另一个常量字符串
printf(“%s\n”, strs[0]);
// 为什么地址的差值为8? 因为指针占用八个字节，反过来印证了数组里面的元素是指针。
printf(“%p\n”, strs);
printf(“%p\n”, strs + 1);
printf(“%p\n”, strs + 2);

指针与函数

/*
 思考 如何用指针 实习两个数交换函数
 */
void changeValue2(int *p1, int *p2);//声明函数
int main(int argc, const char * argv[]) {
    int a1 = 10, b1 = 20;
    changeValue2(&a1, &b1);
    printf("%d, %d",a1,b1); //验证值是否替换
    return 0;
}
void changeValue2(int *p1, int *p2)
{
    int temp = *p1; // 取出p1所指向的地址里面的数据
    *p1 = *p2;
    *p2 = temp;
    printf("%d   %d\n", *p1, *p2); //验证是否地址的值被替换
}
//利用函数传入地址交换数值
